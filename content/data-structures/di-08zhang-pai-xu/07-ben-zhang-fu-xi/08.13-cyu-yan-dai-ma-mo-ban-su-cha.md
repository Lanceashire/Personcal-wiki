## 一、直接插入排序

```c
void insertionSort(int arr[], int n) {
    for (int i = 1; i < n; i++) {
        int temp = arr[i];
        int j = i - 1;
        while (j >= 0 && arr[j] > temp) {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = temp;
    }
}
```

## 二、希尔排序

```c
void shellSort(int arr[], int n) {
    for (int gap = n / 2; gap > 0; gap /= 2) {
        for (int i = gap; i < n; i++) {
            int temp = arr[i];
            int j = i;
            while (j >= gap && arr[j - gap] > temp) {
                arr[j] = arr[j - gap];
                j -= gap;
            }
            arr[j] = temp;
        }
    }
}
```

## 三、冒泡排序

```c
void bubbleSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int swapped = 0;
        for (int j = 0; j < n - 1 - i; j++) {
            if (arr[j] > arr[j + 1]) {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
                swapped = 1;
            }
        }
        if (!swapped) break;
    }
}
```

## 四、快速排序

```c
void quickSort(int arr[], int left, int right) {
    if (left >= right) return;
    int i = left, j = right;
    int pivot = arr[left];
    while (i < j) {
        while (i < j && arr[j] >= pivot) j--;
        while (i < j && arr[i] <= pivot) i++;
        if (i < j) {
            int temp = arr[i]; arr[i] = arr[j]; arr[j] = temp;
        }
    }
    arr[left] = arr[i]; arr[i] = pivot;
    quickSort(arr, left, i - 1);
    quickSort(arr, i + 1, right);
}
```

## 五、简单选择排序

```c
void selectionSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int minIndex = i;
        for (int j = i + 1; j < n; j++) {
            if (arr[j] < arr[minIndex]) minIndex = j;
        }
        if (minIndex != i) {
            int temp = arr[i]; arr[i] = arr[minIndex]; arr[minIndex] = temp;
        }
    }
}
```

## 六、堆排序

```c
void heapify(int arr[], int n, int root) {
    int largest = root;
    int left = 2 * root + 1;
    int right = 2 * root + 2;
    if (left < n && arr[left] > arr[largest]) largest = left;
    if (right < n && arr[right] > arr[largest]) largest = right;
    if (largest != root) {
        int temp = arr[root]; arr[root] = arr[largest]; arr[largest] = temp;
        heapify(arr, n, largest);
    }
}

void heapSort(int arr[], int n) {
    for (int i = n / 2 - 1; i >= 0; i--) heapify(arr, n, i);
    for (int end = n - 1; end > 0; end--) {
        int temp = arr[0]; arr[0] = arr[end]; arr[end] = temp;
        heapify(arr, end, 0);
    }
}
```

## 七、归并排序

```c
#include <stdlib.h>

void merge(int arr[], int left, int mid, int right) {
    int len = right - left + 1;
    int *temp = malloc(len * sizeof(int));
    if (temp == NULL) return;
    int i = left, j = mid + 1, k = 0;
    while (i <= mid && j <= right) {
        if (arr[i] <= arr[j]) temp[k++] = arr[i++];
        else temp[k++] = arr[j++];
    }
    while (i <= mid) temp[k++] = arr[i++];
    while (j <= right) temp[k++] = arr[j++];
    for (int p = 0; p < len; p++) arr[left + p] = temp[p];
    free(temp);
}

void mergeSort(int arr[], int left, int right) {
    if (left >= right) return;
    int mid = left + (right - left) / 2;
    mergeSort(arr, left, mid);
    mergeSort(arr, mid + 1, right);
    merge(arr, left, mid, right);
}
```

## 十、默写时的检查清单

### 直接插入

- [ ] 是否保存 `temp`？
- [ ] 是否从 `i = 1` 开始？
- [ ] 是否插入到 `j + 1`？

### 冒泡

- [ ] 是否访问 `arr[j + 1]` 时避免越界？
- [ ] 是否减去已排好尾部长度？
- [ ] 是否有 `swapped`？

### 快速排序

- [ ] 是否有递归结束条件？
- [ ] 是否保存 `pivot`？
- [ ] 左右指针是否都会收缩？
- [ ] 是否递归处理基准两侧？

### 选择排序

- [ ] 是否先完整找出 `minIndex`？
- [ ] 是否每趟最多交换一次？

### 堆排序

- [ ] 左孩子是否为 `2 * root + 1`？
- [ ] 右孩子是否为 `2 * root + 2`？
- [ ] 交换堆顶后，调整长度是否缩小？

### 归并排序

- [ ] 是否复制两侧剩余元素？
- [ ] 是否将临时数组复制回原数组？
- [ ] 是否释放内存？
