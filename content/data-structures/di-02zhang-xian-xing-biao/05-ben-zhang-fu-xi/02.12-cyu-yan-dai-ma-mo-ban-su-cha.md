## 1. 顺序表定义

```c
#include <stdio.h>
#include <stdlib.h>

#define MAX_SIZE 100

typedef int ElemType;

typedef struct {
    ElemType data[MAX_SIZE];
    int length;
} SeqList;
```

## 2. 顺序表插入

```c
int seqInsert(
    SeqList *list,
    int index,
    ElemType value
) {
    if (
        list == NULL ||
        index < 0 ||
        index > list->length ||
        list->length >= MAX_SIZE
    ) {
        return 0;
    }

    for (int j = list->length; j > index; j--) {
        list->data[j] = list->data[j - 1];
    }

    list->data[index] = value;
    list->length++;
    return 1;
}
```

## 3. 顺序表删除

```c
int seqDelete(
    SeqList *list,
    int index,
    ElemType *removed
) {
    if (
        list == NULL ||
        index < 0 ||
        index >= list->length
    ) {
        return 0;
    }

    if (removed != NULL) {
        *removed = list->data[index];
    }

    for (
        int j = index;
        j < list->length - 1;
        j++
    ) {
        list->data[j] = list->data[j + 1];
    }

    list->length--;
    return 1;
}
```

## 4. 单链表定义与创建

```c
typedef struct Node {
    ElemType data;
    struct Node *next;
} Node;

Node *createNode(ElemType value) {
    Node *node = malloc(sizeof(Node));

    if (node == NULL) {
        return NULL;
    }

    node->data = value;
    node->next = NULL;
    return node;
}

Node *createList(void) {
    return createNode(0);
}
```

## 5. 单链表按位置插入

```c
int listInsert(
    Node *head,
    int index,
    ElemType value
) {
    if (head == NULL || index < 0) {
        return 0;
    }

    Node *previous = head;

    for (int i = 0; i < index; i++) {
        if (previous->next == NULL) {
            return 0;
        }

        previous = previous->next;
    }

    Node *newNode = createNode(value);

    if (newNode == NULL) {
        return 0;
    }

    newNode->next = previous->next;
    previous->next = newNode;
    return 1;
}
```

## 6. 单链表按位置删除

```c
int listDelete(
    Node *head,
    int index,
    ElemType *removed
) {
    if (head == NULL || index < 0) {
        return 0;
    }

    Node *previous = head;

    for (int i = 0; i < index; i++) {
        if (previous->next == NULL) {
            return 0;
        }

        previous = previous->next;
    }

    Node *target = previous->next;

    if (target == NULL) {
        return 0;
    }

    if (removed != NULL) {
        *removed = target->data;
    }

    previous->next = target->next;
    free(target);
    return 1;
}
```

## 7. 尾插法建表

```c
Node *createByTail(
    const ElemType values[],
    int n
) {
    if (values == NULL || n < 0) {
        return NULL;
    }

    Node *head = createList();

    if (head == NULL) {
        return NULL;
    }

    Node *tail = head;

    for (int i = 0; i < n; i++) {
        Node *newNode = createNode(values[i]);

        if (newNode == NULL) {
            destroyList(head);
            return NULL;
        }

        tail->next = newNode;
        tail = newNode;
    }

    return head;
}
```

在同一 `.c` 文件中，应在调用前声明：

```c
void destroyList(Node *head);
```

## 8. 链表销毁

```c
void destroyList(Node *head) {
    Node *current = head;

    while (current != NULL) {
        Node *nextNode = current->next;
        free(current);
        current = nextNode;
    }
}
```

## 9. 有序链表原地合并

```c
Node *mergeSorted(
    Node *headA,
    Node *headB
) {
    if (headA == NULL) return headB;
    if (headB == NULL) return headA;

    Node *pa = headA->next;
    Node *pb = headB->next;
    Node *tail = headA;

    while (pa != NULL && pb != NULL) {
        if (pa->data <= pb->data) {
            tail->next = pa;
            pa = pa->next;
        } else {
            tail->next = pb;
            pb = pb->next;
        }

        tail = tail->next;
    }

    tail->next = (pa != NULL) ? pa : pb;
    free(headB);
    return headA;
}
```

## 10. 默写检查

- [ ] 插入前检查范围与容量。
- [ ] 顺序表插入从后向前移动。
- [ ] 顺序表删除从前向后移动。
- [ ] 链表插入先让新结点接后继。
- [ ] 链表删除先保存目标结点。
- [ ] 每次 `malloc` 后检查 `NULL`。
- [ ] 释放当前结点前保存 `next`。
- [ ] 原地合并后明确原链表的所有权变化。
