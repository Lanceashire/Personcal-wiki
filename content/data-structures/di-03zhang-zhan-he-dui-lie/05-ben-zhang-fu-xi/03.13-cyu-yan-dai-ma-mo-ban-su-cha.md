## 1. 顺序栈

```c
#define CAPACITY 100

typedef int ElemType;

typedef struct {
    ElemType data[CAPACITY];
    int top;
} SeqStack;

void initStack(SeqStack *s) {
    s->top = -1;
}

int push(SeqStack *s, ElemType value) {
    if (s == NULL || s->top >= CAPACITY - 1) {
        return 0;
    }

    s->data[++s->top] = value;
    return 1;
}

int pop(SeqStack *s, ElemType *result) {
    if (s == NULL || s->top < 0) {
        return 0;
    }

    if (result != NULL) {
        *result = s->data[s->top];
    }

    s->top--;
    return 1;
}
```

## 2. 链栈

```c
typedef struct StackNode {
    ElemType data;
    struct StackNode *next;
} StackNode;

typedef struct {
    StackNode *top;
} LinkedStack;

int linkedPush(LinkedStack *s, ElemType value) {
    if (s == NULL) return 0;

    StackNode *node = malloc(sizeof(StackNode));
    if (node == NULL) return 0;

    node->data = value;
    node->next = s->top;
    s->top = node;
    return 1;
}

int linkedPop(LinkedStack *s, ElemType *result) {
    if (s == NULL || s->top == NULL) return 0;

    StackNode *target = s->top;
    if (result != NULL) *result = target->data;
    s->top = target->next;
    free(target);
    return 1;
}
```

## 3. 循环队列

约定：

```text
front 指向队头
rear 指向下一空位
牺牲一个数组位置
```

```c
typedef struct {
    ElemType data[CAPACITY];
    int front;
    int rear;
} CircularQueue;

void initQueue(CircularQueue *q) {
    q->front = 0;
    q->rear = 0;
}

int queueEmpty(const CircularQueue *q) {
    return q->front == q->rear;
}

int queueFull(const CircularQueue *q) {
    return (q->rear + 1) % CAPACITY == q->front;
}

int enQueue(CircularQueue *q, ElemType value) {
    if (q == NULL || queueFull(q)) return 0;

    q->data[q->rear] = value;
    q->rear = (q->rear + 1) % CAPACITY;
    return 1;
}

int deQueue(CircularQueue *q, ElemType *result) {
    if (q == NULL || queueEmpty(q)) return 0;

    if (result != NULL) {
        *result = q->data[q->front];
    }

    q->front = (q->front + 1) % CAPACITY;
    return 1;
}

int queueLength(const CircularQueue *q) {
    return
        (q->rear - q->front + CAPACITY) %
        CAPACITY;
}
```

## 4. 链队列

```c
typedef struct QueueNode {
    ElemType data;
    struct QueueNode *next;
} QueueNode;

typedef struct {
    QueueNode *front;
    QueueNode *rear;
} LinkedQueue;

int initLinkedQueue(LinkedQueue *q) {
    if (q == NULL) return 0;

    QueueNode *dummy = malloc(sizeof(QueueNode));
    if (dummy == NULL) return 0;

    dummy->next = NULL;
    q->front = dummy;
    q->rear = dummy;
    return 1;
}

int linkedEnQueue(LinkedQueue *q, ElemType value) {
    if (q == NULL) return 0;

    QueueNode *node = malloc(sizeof(QueueNode));
    if (node == NULL) return 0;

    node->data = value;
    node->next = NULL;
    q->rear->next = node;
    q->rear = node;
    return 1;
}

int linkedDeQueue(LinkedQueue *q, ElemType *result) {
    if (q == NULL || q->front == q->rear) return 0;

    QueueNode *target = q->front->next;
    if (result != NULL) *result = target->data;
    q->front->next = target->next;

    if (q->rear == target) {
        q->rear = q->front;
    }

    free(target);
    return 1;
}
```

## 5. 默写检查

- [ ] 顺序栈 `top` 约定写在代码旁。
- [ ] 入栈前判满，出栈前判空。
- [ ] 链栈先连接旧栈顶，再移动 `top`。
- [ ] 循环队列使用模运算。
- [ ] 循环队列说明是否牺牲一个位置。
- [ ] 链队列删除最后结点时重置 `rear`。
- [ ] 每次 `malloc` 检查返回值。
- [ ] 释放结点前保存所需链接。
