推荐考试和刷题优先掌握前缀函数版 KMP。教材若指定 next[0]=-1 或 1 下标 next，再使用对应模板。

> [!summary]
> 推荐考试和刷题优先掌握前缀函数版 KMP。教材若指定 `next[0]=-1` 或 1 下标 `next`，再使用对应模板。

上级知识点：本章目录

## 1. 前缀函数

```c
void buildPi(
    const char *pattern,
    int m,
    int pi[]
) {
    if (
        pattern == NULL ||
        pi == NULL ||
        m <= 0
    ) {
        return;
    }

    pi[0] = 0;

    for (int i = 1; i < m; i++) {
        int k = pi[i - 1];

        while (
            k > 0 &&
            pattern[i] != pattern[k]
        ) {
            k = pi[k - 1];
        }

        if (pattern[i] == pattern[k]) {
            k++;
        }

        pi[i] = k;
    }
}
```

## 2. 前缀函数版 KMP

```c
int kmpSearch(
    const char *text,
    int n,
    const char *pattern,
    int m
) {
    if (
        text == NULL ||
        pattern == NULL ||
        n < 0 ||
        m < 0
    ) {
        return -1;
    }

    if (m == 0) {
        return 0;
    }

    int *pi = malloc((size_t)m * sizeof(int));

    if (pi == NULL) {
        return -1;
    }

    buildPi(pattern, m, pi);
    int j = 0;

    for (int i = 0; i < n; i++) {
        while (
            j > 0 &&
            text[i] != pattern[j]
        ) {
            j = pi[j - 1];
        }

        if (text[i] == pattern[j]) {
            j++;
        }

        if (j == m) {
            int position = i - m + 1;
            free(pi);
            return position;
        }
    }

    free(pi);
    return -1;
}
```

## 3. 查找全部出现

成功后不要返回，记录：

```c
int position = i - m + 1;
```

然后：

```c
j = pi[m - 1];
```

继续扫描，才能保留重叠匹配。

## 4. 0 下标 `next[0]=-1`

```c
void buildNext(
    const char *pattern,
    int m,
    int next[]
) {
    if (m <= 0) return;

    int j = 0;
    int k = -1;
    next[0] = -1;

    while (j < m - 1) {
        if (
            k == -1 ||
            pattern[j] == pattern[k]
        ) {
            j++;
            k++;
            next[j] = k;
        } else {
            k = next[k];
        }
    }
}
```

搜索：

```c
int searchWithNext(
    const char *text,
    int n,
    const char *pattern,
    int m,
    const int next[]
) {
    if (m == 0) return 0;

    int i = 0;
    int j = 0;

    while (i < n && j < m) {
        if (
            j == -1 ||
            text[i] == pattern[j]
        ) {
            i++;
            j++;
        } else {
            j = next[j];
        }
    }

    return j == m ? i - j : -1;
}
```

## 5. `nextval`

```c
void buildNextVal(
    const char *pattern,
    int m,
    int nextVal[]
) {
    if (m <= 0) return;

    int j = 0;
    int k = -1;
    nextVal[0] = -1;

    while (j < m - 1) {
        if (
            k == -1 ||
            pattern[j] == pattern[k]
        ) {
            j++;
            k++;

            if (pattern[j] != pattern[k]) {
                nextVal[j] = k;
            } else {
                nextVal[j] = nextVal[k];
            }
        } else {
            k = nextVal[k];
        }
    }
}
```

搜索函数与普通 `next` 相同，只需传入 `nextVal`。

## 6. 重复周期判断

```c
int isRepeatedPattern(
    const char *s,
    int n
) {
    if (s == NULL || n <= 1) {
        return 0;
    }

    int *pi = malloc((size_t)n * sizeof(int));
    if (pi == NULL) return 0;

    buildPi(s, n, pi);

    int border = pi[n - 1];
    int period = n - border;
    int result =
        border > 0 &&
        n % period == 0;

    free(pi);
    return result;
}
```

## 7. 默写检查

- [ ] `pi[0] = 0`。
- [ ] 构造失配时 `k = pi[k-1]`。
- [ ] 搜索失配前检查 `j > 0`。
- [ ] 成功位置为 `i-m+1`。
- [ ] 空模式单独处理。
- [ ] `next[0]=-1` 模板与 `pi` 模板不混用。
- [ ] 查找全部匹配后 `j=pi[m-1]`。
- [ ] 动态数组在所有返回路径上释放。
